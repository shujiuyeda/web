⑨ 「今日もコレカラ」バナー自動化スクリプト（update_banner.sh）

初回: 現在のVol番号をセット（例: 683）
echo "683" > sns/vol_number.txt
スクリプト本体（scripts/update_banner.sh に保存）
#!/bin/bash
# 使い方: bash scripts/update_banner.sh
# Vol番号は sns/vol_number.txt から自動取得・自動インクリメント

VOLFILE="sns/vol_number.txt"
AIFILE="$HOME/Desktop/今日もコレカラバナー.ai"  # ← 実際のパスに変更

VOL=$(cat "$VOLFILE")
NEXT_VOL=$((VOL + 1))
echo "$NEXT_VOL" > "$VOLFILE"

DATE=$(date '+%Y/%-m/%-d')
OUTFILE="sns/今日もコレカラ_$(date '+%Y%m%d').png"

osascript <<EOF
tell application "Adobe Illustrator"
    open POSIX file "$AIFILE"
    set doc to current document
    repeat with tf in every text frame of doc
        if text of tf contains "Vol." then
            set text of tf to "Vol.$VOL　$DATE"
        end if
    end repeat
    export doc to POSIX file "$(pwd)/$OUTFILE" as PNG
    close doc saving no
end tell
EOF

echo "書き出し完了: $OUTFILE（Vol.$VOL / $DATE）"
実行（毎日これだけ）
bash scripts/update_banner.sh
